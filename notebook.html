import matplotlib as plt

corr = np.corrcoef(np.random.randn(12, 150))

mask = np.zeros_like(corr)

mask[np.triu_indices_from(mask)] = True

with sns.axes_style("white"):

    f, ax = plt.pyplot.subplots(figsize=(14, 12))
    sns.set(rc={'figure.figsize':(14,12)})
    ax = sns.heatmap(df_white.corr(), mask=mask, vmax=.3, square=True, cmap="RdBu_r", annot = True, fmt='.2f')

figure = ax.get_figure()    
figure.savefig('whiteWine_correlation.png')
